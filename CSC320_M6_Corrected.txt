public String updateHome(String homeModel, String homeAddress, int zipCode) {
    try {
        for (Home home : homes) {
            if (home.getHomeModel().equals(homeModel) && home.getZipCode() == zipCode) {
                home.setHomeAddress(homeAddress);
                return "Home successfully updated.";
            }
        }
        return "Home not found in inventory.";
    } catch (Exception e) {
        return "Error updating home: " + e.getMessage();
    }
}

REMOVE A HOME:

public String removeHome(String homeModel, String homeAddress, int zipCode) {
    try {
        for (int i = 0; i < homes.size(); i++) {
            Home home = homes.get(i);

            if (home.getHomeModel().equals(homeModel) &&
                home.getHomeAddress().equals(homeAddress) &&
                home.getZipCode() == zipCode) {

                homes.remove(i);
                return "Home successfully removed from inventory.";
            }
        }
        return "Home not found in inventory.";
    } catch (Exception e) {
        return "Error removing home: " + e.getMessage();
    }
}
